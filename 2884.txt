#include <stdio.h>
/*
주어진 시간보다 45분 앞에 시간 구하기
1. 45분 이상 일 때
  - -45분 저장
2. 45분 이하 일 때
 - 60+원래 분 - 45를 저장한다 - 몇분인지에 쓰임
3. 45분 이하 이면 '시'를 1내림
 - 만약 원래 0시였다면 23시로 바꿈 
*/

int main(void) {
  int h, m;

  scanf("%d %d", &h, &m);

  if(m >= 45){
    m -= 45;
  }else{
    m += 60 - 45;
    if(h != 0){
      h -= 1;
    }else{
      h = 23;
    }
  }

  printf("%d %d", h, m);

  return 0;
}
